1. Install Git and configure username and email.

#  Step 1: Update package list to ensure latest versions are available
sudo apt update

#  Step 2: Install Git using the package manager
sudo apt install git -y

#  Step 3: Verify Git installation
git --version

#  Step 4: Set your Git username (replace "Your Name" with your actual name)
git config --global user.name "Your Name"

# Step 5: Set your Git email (replace with your actual email)
git config --global user.email "you@example.com"

# Step 6: Confirm that the configuration was successful
git config --global --list


2. Create a local repository, add a file, and commit changes.

#  Step 1: Create a new directory for your project
mkdir my_project

#  Step 2: Move into the project directory
cd my_project

# Step 3: Initialize a new Git repository
git init

# Step 4: Create a new file (e.g., README.txt)
echo "This is my first Git project." > README.txt

# Step 5: Add the file to the staging area
git add README.txt

# Step 6: Commit the file with a message
git commit -m "Initial commit: added README.txt"

#Step 7: Check the status of your repository
git status

# Step 8: View the commit history
git log


3. Create a new branch, switch to it, make changes, and merge it with `main`.

# Step 1: Create a new branch named 'feature-branch'
git branch feature-branch

# Step 2: Switch to the new branch
git checkout feature-branch

# Step 3: Make changes â€” for example, edit or create a file
echo "This is a new feature." >> feature.txt

# Step 4: Add the changes to staging
git add feature.txt

# Step 5: Commit the changes with a message
git commit -m "Added feature.txt in feature-branch"

# Step 6: Switch back to the main branch
git checkout main

# Step 7: Merge the feature branch into main
git merge feature-branch

# Step 8: (Optional) Delete the feature branch after merging
git branch -d feature-branch


4. Push the local repository to GitHub using command line.

# Step 1: Create a new repository on GitHub (do this in your browser first)
# Example: https://github.com/your-username/my_project

# Step 2: Add the remote URL to your local repo (replace with your actual GitHub URL)
git remote add origin https://github.com/your-username/my_project.git

# Step 3: Push your local commits to GitHub (main branch)
git push -u origin main


5. Clone a repository from GitHub, edit files, and push changes back.

# Step 1: Clone the repository (replace with your actual GitHub repo URL)
git clone https://github.com/your-username/your-repo.git

# Step 2: Move into the cloned repository directory
cd your-repo

# Step 3: Edit or create a file (example: add a line to README.md)
echo "New update from local machine." >> README.md

# Step 4: Add the modified file to the staging area
git add README.md

# Step 5: Commit the changes with a message
git commit -m "Updated README.md with a new line"

# Step 6: Push the changes back to GitHub
git push origin main
